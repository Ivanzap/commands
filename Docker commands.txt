  -- docker images | grep имя_образа - остортирует образы по имени
  
  -- docker-compose up --build - сначала сбилдит образ, а потом запустит приложение
  
  ---------------------------------------------------------------
  docker-compose.yml

  -- version: '3' - версия docker-compose и какие поля можно использовать внутри этого файла
  
  -- services: - сервисы для backend'a, frontend'a и т.д
  
  services:
    api: - сервис api
      build: ./api - команда, которая указывает, что docker-compose должен взять Dockerfile из папки api
      command: npm run start - которая выполнится при старте сервиса
      restart: unless-stopped - если произойдет ошибка или контейнер упадет, то docker перезапустит контейнер
      ports:
        - "3000:3000" - открываем для наружи порт для 3000, слева хост, справа порт контейнера
      envionment:
        - PORT=3000 - указали новую переменную окружения
   
  ---------------------------------------------------------------
  Dockerfile - нужен для создание образов. Означает, как мы будем билдить наш образ
  В Docker'e пишем какой образ мы хотим создать.
  Dockerfile отвечает за то, как сервис будет собираться в docker image
  
  -- FROM node:13 - создаем образ api на основе образа node версии последней версии из 13
  -- FROM node:13.12.0-alpine - alpine меньше занимает места, лучше использовать alpine если имеется
  
  -- WORKDIR /usr/src/app - указываем рабочую дирректорию, внутри контейнера создается доп папка, в которой будем работать
  
  -- COPY имя_файла ./ - копирует все файлы с локального репозитория в докер контейнер 
    -- COPY package*.json ./ - копирует файлы с частью имени package и формата json в корневую папку (./), а именно в /usr/src/app
  
  -- RUN npm install - запуск nmp install (установка пакетов внутри образа) Внутри контейнера
  
  -- COPY . . - скопировать все файлы внутрь /usr/src/app
  
  -- EXPOSE 3000 - порт на котором запускаем приложение из докер контейнера для внешнего мира
  
  -- CMD ["node", "run start"]
  
  Docker image - это образ linux, и с него устанавливаем контейнер
